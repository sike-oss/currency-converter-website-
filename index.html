from pyscript import Element def usd_to_inr(usd): return usd * 82.73 def eur_to_inr(eur): return eur * 90.12 def aed_to_inr(aed): return aed * 22.49 def gbp_to_inr(gbp): return gbp * 102.34 def convert_currency(event=None): choice = int(Element("choice").element.value) amount_input = Element("amount").element.value output = Element("output") if choice == 5: output.write("Exiting the converter. Refresh to restart.") return if not amount_input: output.write("⚠️ Please enter an amount.") return try: amount = float(amount_input) except: output.write("⚠️ Invalid amount entered.") return if choice == 1: result = f"{amount} USD is equal to {usd_to_inr(amount):.2f} INR" elif choice == 2: result = f"{amount} EUR is equal to {eur_to_inr(amount):.2f} INR" elif choice == 3: result = f"{amount} AED is equal to {aed_to_inr(amount):.2f} INR" elif choice == 4: result = f"{amount} GBP is equal to {gbp_to_inr(amount):.2f} INR" else: result = "Invalid choice. Please enter a number between 1 and 5." output.write(result) # Connect button Element("convert").element.onclick = convert_currency
